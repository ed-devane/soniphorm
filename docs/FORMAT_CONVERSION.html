JSON Format Conversion Guide ​

CodegenApp Format → Runtime Format ​

The current CodegenApp JSON format is 70% compatible with the GenRuntime format.

Differences ​

FieldCodegenAppGenRuntimeConversion Needed?
Top-level✅ Same✅ SameNo
Module positionx, y flatposition: {x, y} nestedYes (minor)
Connection IDsfrom.id: "osc1:outL"from.module: "osc1", from.output: "outL"Yes (major)
Connection fieldsfrom, to objects with id, port, kindfrom, to objects with module, output/inputYes (major)

JavaScript Conversion Function ​

Add this to CodegenApp/src/app.jsx:

javascript
const convertPatchToRuntime = (codegenPatch) => {
  return {
    version: codegenPatch.version || "1.0",
    name: codegenPatch.name,
    description: codegenPatch.description || "",
    author: codegenPatch.author || "",
    created: codegenPatch.created,

    // Convert modules - nest x/y in position object
    modules: codegenPatch.modules.map(m => ({
      id: m.id,
      type: m.type,
      position: {
        x: m.x || 100,
        y: m.y || 100
      },
      params: m.params
    })),

    // Convert connections - split combined ID format
    connections: codegenPatch.connections.map(c => {
      // Parse "moduleId:portName" format
      const fromParts = c.from.id.split(':');
      const toParts = c.to.id.split(':');

      // Determine if port is input or output
      const isFromOutput = c.from.port; // Outputs are in "from"
      const isToInput = c.to.port;     // Inputs are in "to"

      return {
        from: {
          module: fromParts[0],
          output: fromParts[1] || c.from.port
        },
        to: {
          module: toParts[0],
          input: toParts[1] || c.to.port
        }
      };
    })
  };
};

// Usage in savePatch function:
const saveRuntimePatch = () => {
  const patchName = prompt("Enter patch name:", "Untitled Patch");
  if (patchName === null) return;

  const sanitizedName = patchName.trim().replace(/[^a-z0-9_\\- ]/gi, '_') || 'Untitled';

  // Create CodegenApp format first
  const codegenPatch = {
    version: "1.0",
    name: patchName,
    created: new Date().toISOString(),
    modules: modules.map(m => ({
      id: m.id,
      type: m.type,
      x: m.x,
      y: m.y,
      params: m.params
    })),
    connections: connections.map(c => ({
      from: c.from,
      to: c.to
    }))
  };

  // Convert to runtime format
  const runtimePatch = convertPatchToRuntime(codegenPatch);

  // Download runtime-compatible JSON
  const jsonString = JSON.stringify(runtimePatch, null, 2);
  const blob = new Blob([jsonString], { type: 'application/json' });
  const url = URL.createObjectURL(blob);
  const link = document.createElement('a');
  link.href = url;
  link.download = `${sanitizedName}_runtime.json`;
  document.body.appendChild(link);
  link.click();
  document.body.removeChild(link);
  URL.revokeObjectURL(url);

  console.log(`Runtime patch saved as: ${sanitizedName}_runtime.json`);
};

Python Conversion Script ​

For batch conversion of existing patches:

python
import json
import sys

def convert_patch(codegen_patch):
    """Convert CodegenApp JSON to Runtime JSON format"""
    runtime_patch = {
        "version": codegen_patch.get("version", "1.0"),
        "name": codegen_patch["name"],
        "description": codegen_patch.get("description", ""),
        "author": codegen_patch.get("author", ""),
        "created": codegen_patch["created"],
        "modules": [],
        "connections": []
    }

    # Convert modules
    for module in codegen_patch["modules"]:
        runtime_module = {
            "id": module["id"],
            "type": module["type"],
            "position": {
                "x": module.get("x", 100),
                "y": module.get("y", 100)
            },
            "params": module.get("params", {})
        }
        runtime_patch["modules"].append(runtime_module)

    # Convert connections
    for conn in codegen_patch["connections"]:
        # Parse "moduleId:portName" from id field
        from_parts = conn["from"]["id"].split(':')
        to_parts = conn["to"]["id"].split(':')

        runtime_conn = {
            "from": {
                "module": from_parts[0],
                "output": from_parts[1] if len(from_parts) > 1 else conn["from"].get("port", "out")
            },
            "to": {
                "module": to_parts[0],
                "input": to_parts[1] if len(to_parts) > 1 else conn["to"].get("port", "in")
            }
        }
        runtime_patch["connections"].append(runtime_conn)

    return runtime_patch

if __name__ == "__main__":
    if len(sys.argv) != 3:
        print("Usage: python convert_patch.py input.json output.json")
        sys.exit(1)

    input_file = sys.argv[1]
    output_file = sys.argv[2]

    with open(input_file, 'r') as f:
        codegen_patch = json.load(f)

    runtime_patch = convert_patch(codegen_patch)

    with open(output_file, 'w') as f:
        json.dump(runtime_patch, f, indent=2)

    print(f"Converted {input_file} -> {output_file}")

  JsonArray connectionsArray = doc["connections"];

    }

  return true;
